static VALUE
nary_robject_s_cast(VALUE type, VALUE obj)
{
    VALUE r;

    if (CLASS_OF(obj)==cT) {
        return obj;
    }
    
    if (FIXNUM_P(obj) || TYPE(obj)==T_FLOAT || TYPE(obj)==T_BIGNUM || rb_obj_is_kind_of(obj,cComplex)) {
        return nary_robject_cast_numeric(obj);
    }
    
    if (TYPE(obj)==T_ARRAY) {
        return nary_robject_cast_array(obj);
    }
    
    if (rb_obj_is_kind_of(obj,cDComplex)) {
        return nary_robject_cast_dcomplex(obj);
    }
    
    if (rb_obj_is_kind_of(obj,cSComplex)) {
        return nary_robject_cast_scomplex(obj);
    }
    
    if (rb_obj_is_kind_of(obj,cDFloat)) {
        return nary_robject_cast_dfloat(obj);
    }
    
    if (rb_obj_is_kind_of(obj,cSFloat)) {
        return nary_robject_cast_sfloat(obj);
    }
    
    if (rb_obj_is_kind_of(obj,cInt64)) {
        return nary_robject_cast_int64(obj);
    }
    
    if (rb_obj_is_kind_of(obj,cInt32)) {
        return nary_robject_cast_int32(obj);
    }
    
    if (rb_obj_is_kind_of(obj,cInt16)) {
        return nary_robject_cast_int16(obj);
    }
    
    if (rb_obj_is_kind_of(obj,cInt8)) {
        return nary_robject_cast_int8(obj);
    }
    
    if (rb_obj_is_kind_of(obj,cUInt64)) {
        return nary_robject_cast_uint64(obj);
    }
    
    if (rb_obj_is_kind_of(obj,cUInt32)) {
        return nary_robject_cast_uint32(obj);
    }
    
    if (rb_obj_is_kind_of(obj,cUInt16)) {
        return nary_robject_cast_uint16(obj);
    }
    
    if (rb_obj_is_kind_of(obj,cUInt8)) {
        return nary_robject_cast_uint8(obj);
    }
    

    if (IsNArray(obj)) {
        r = rb_funcall(obj, rb_intern("coerce_cast"), 1, cT);
        if (RTEST(r)) {
            return r;
        }
    }
    rb_raise(nary_eCastError, "unknown conversion from %s to %s",
             rb_class2name(CLASS_OF(obj)),
             rb_class2name(type));
    return r;
}

static VALUE
nary_robject_seq(int argc, VALUE *args, VALUE self)
{
    seq_opt_t *g;
    VALUE vbeg=Qnil, vstep=Qnil;
    ndfunc_t *func;

    func = ndfunc_alloc(iter_robject_seq, FULL_LOOP,
                        1, 0, Qnil);
    g = ALLOCA_N(seq_opt_t,1);
    g->beg = 0;
    g->step = 1;
    g->count = 0;
    rb_scan_args(argc, args, "02", &vbeg, &vstep);
    if (vbeg!=Qnil) {g->beg = NUM2DBL(vbeg);}
    if (vstep!=Qnil) {g->step = NUM2DBL(vstep);}
    ndloop_do3(func, g, 1, self);
    ndfunc_free(func);
    return self;
}
